# Media Tools - Requirements
# ===========================
# Dependências Python para processamento de mídia
#
# Python 3.6+ é necessário
#
# Para instalar todas as dependências:
#   pip install -r requirements.txt
#
# Para instalar em ambiente virtual (recomendado):
#   python -m venv venv
#   source venv/bin/activate  # Linux/macOS
#   .\venv\Scripts\activate   # Windows
#   pip install -r requirements.txt

# ============================================
# DEPENDÊNCIA EXTERNA NECESSÁRIA
# ============================================
# FFmpeg: Deve estar instalado no sistema e disponível no PATH
#   - Necessário apenas para scripts de vídeo (otimizador-video.py, webm-mp4.py)
#   - Download: https://ffmpeg.org/download.html
#
# Instalação:
#   Windows: https://www.gyan.dev/ffmpeg/builds/ ou via chocolatey: choco install ffmpeg
#   Linux: sudo apt-get install ffmpeg (Ubuntu/Debian) ou sudo yum install ffmpeg (CentOS/RHEL)
#   macOS: brew install ffmpeg
#
# pngquant: Opcional - para compressão adicional de PNGs
#   Windows: https://pngquant.org/ ou via chocolatey: choco install pngquant
#   Linux: sudo apt-get install pngquant
#   macOS: brew install pngquant

# ============================================
# PROCESSAMENTO DE IMAGENS
# ============================================
# Pillow: Biblioteca para manipulação de imagens
#   - Usado em: otimizador-imagens.py, webp-to-jpg.py
Pillow>=10.2.0

# pillow-heif: Suporte para HEIC/HEIF
#   - Usado em: otimizador-imagens.py (para converter HEIC/HEIF)
pillow-heif>=0.13.0

# ============================================
# PROCESSAMENTO DE VÍDEOS E ANÁLISE DE IMAGENS
# ============================================
# OpenCV: Biblioteca para processamento de imagens e vídeos
#   - Usado em: validate-images.py, estabilizador-video.py
opencv-python>=4.12.0

# NumPy: Biblioteca para computação numérica
#   - Usado em: validate-images.py, estabilizador-video.py (análise de imagens)
#   - Nota: Python 3.14 é muito novo, pode não ter wheels pré-compiladas
#   - Se falhar, instale manualmente: pip install numpy (sem versão específica)
#   - Ou use Python 3.11/3.12 que têm melhor suporte de wheels
numpy

# ============================================
# OCR E PROCESSAMENTO AVANÇADO
# ============================================
# pytesseract: OCR para detecção de texto em imagens
#   - Usado em: ocr-imagens.py
#   - Requer Tesseract OCR instalado no sistema
#   - Windows: https://github.com/UB-Mannheim/tesseract/wiki
#   - Linux: sudo apt-get install tesseract-ocr tesseract-ocr-por
#   - macOS: brew install tesseract tesseract-lang
pytesseract>=0.3.10

# rembg: Remoção automática de fundo
#   - Usado em: remover-fundo.py
#   - Nota: A primeira execução baixará o modelo (~170MB)
rembg>=2.0.50

# ============================================
# UTILITÁRIOS
# ============================================
# tqdm: Barras de progresso em tempo real
#   - Usado em: todos os scripts
tqdm>=4.66.1

# python-dotenv: Carregamento de variáveis de ambiente
#   - Usado para configurações opcionais via .env
python-dotenv>=1.0.0

# requests: Biblioteca HTTP para requisições
#   - Usado para funcionalidades futuras (downloads, APIs, etc.)
requests>=2.31.0

# ============================================
# CONTROLE DE RECURSOS
# ============================================
# psutil: Monitoramento de sistema e processos
#   - Usado em: media_tools/common/resource_control.py (controle de CPU/memória)
psutil>=5.9.0

# imagehash: Hashing de imagens para detecção de duplicatas
#   - Usado em: detector-duplicatas-imagens.py
imagehash>=4.3.1
